  <script>
    function updateMission() {
      const school = document.getElementById('schoolName').value || '[Your School]';
      const action = document.getElementById('action1').value;
      const outcome = document.getElementById('outcome').value;
      const date = document.getElementById('targetDate').value;
      const dateStr = date ? new Date(date).toLocaleDateString('en-GB', {day:'numeric', month:'long', year:'numeric'}) : '[choose a date]';
      
      document.getElementById('missionDisplay').innerHTML = 
        `<span style="color: #2d8a3e; font-weight: 700;">We are ${school}</span> and we pledge to 
         <span style="color: #2d8a3e; font-weight: 700;">${action}</span> so that we can 
         <span style="color: #2d8a3e; font-weight: 700;">${outcome}</span> by 
         <span style="color: #2d8a3e; font-weight: 700;">${dateStr}</span>.`;
    }
    
    function submitMission() {
      const school = document.getElementById('schoolName').value;
      const date = document.getElementById('targetDate').value;
      if (!school || !date) {
        alert('Please fill in your school name and choose a target date!');
        return;
      }
      alert('🎉 Mission submitted! Your pledge has been saved.');
    }
    
    function printMission() {
      const content = document.getElementById('missionDisplay').innerText;
      const school = document.getElementById('schoolName').value || 'Our School';
      const printWindow = window.open('', '_blank');
      printWindow.document.write(`
        <html>
        <head>
          <title>Eco-Hero Mission - ${school}</title>
          <style>
            body { font-family: Arial; display: flex; justify-content: center; align-items: center; min-height: 100vh; margin: 0; background: #e8f5e9; }
            .poster { background: white; padding: 60px; border-radius: 20px; max-width: 600px; text-align: center; box-shadow: 0 10px 40px rgba(0,0,0,0.1); }
            .poster h1 { color: #2d8a3e; font-size: 2rem; margin-bottom: 10px; }
            .poster .mission { font-size: 1.5rem; line-height: 1.8; color: #333; margin: 30px 0; font-style: italic; }
            .poster .highlight { color: #2d8a3e; font-weight: bold; }
            .poster .footer { color: #888; margin-top: 30px; font-size: 0.9rem; }
            .poster .logo { font-size: 4rem; margin-bottom: 10px; }
            @media print { body { background: white; } .poster { box-shadow: none; } }
          </style>
        </head>
        <body>
          <div class="poster">
            <div class="logo">🌿⚡</div>

          </div>
        </body>

      `);
      printWindow.document.close();
      printWindow.print();
    }
    
    document.getElementById('schoolName').addEventListener('input', updateMission);
